	def solution(board, moves):
    answer = 0
    basket=[]
    for i in range(0,len(moves)):
        temp=moves[i]
        for j in range(0,len(board)):
            if(board[j][temp-1]!=0):
                basket.append(board[j][temp-1])
                board[j][temp-1]=0
                if(len(basket)>=2 and basket[-1]==basket[-2]):
                    answer+=2
                    basket=basket[:-2]
                break 
    return answer
